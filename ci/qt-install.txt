function Controller() {
    installer.autoRejectMessageBoxes();
    installer.installationFinished.connect(function() {
        gui.clickButton(buttons.NextButton);
    })
}

Controller.prototype.WelcomePageCallback = function() {
    gui.currentPageWidget().completeChanged.connect(function() {
        gui.clickButton(buttons.NextButton);
    });
}

Controller.prototype.CredentialsPageCallback = function() {
    gui.clickButton(buttons.NextButton);
}

Controller.prototype.IntroductionPageCallback = function() {
    gui.clickButton(buttons.NextButton);
}

Controller.prototype.TargetDirectoryPageCallback = function()
{
    gui.currentPageWidget().TargetDirectoryLineEdit.setText(installer.environmentVariable("QT_INSTALL_DIR"));
    gui.clickButton(buttons.NextButton);
}

Controller.prototype.ComponentSelectionPageCallback = function() {
    console.log("Component Selection Page");

    var components = installer.components();
    console.log("Available packages: " + components.length);
    var packages = ["===LIST OF PACKAGES==="];
    for (var i = 0 ; i < components.length ;i++) {
        packages.push(components[i].name + "    " + components[i].displayName);
    }
    packages.push("===END OF PACKAGES===");
    console.log(packages.join("\n"));

    var widget = gui.currentPageWidget();
    widget.deselectAll();    
    widget.selectComponent("qt.qt5.5123.win64_msvc2017_64");
    widget.deselectComponent("qt.qt5.5123.doc");
    widget.deselectComponent("qt.qt5.5123.examples");
    widget.deselectComponent("qt.qt5.5123.qtcharts");
    widget.deselectComponent("qt.qt5.5123.qtdatavis3d");
    widget.deselectComponent("qt.qt5.5123.qtpurchasing");
    widget.deselectComponent("qt.qt5.5123.qtvirtualkeyboard");
    widget.deselectComponent("qt.qt5.5123.qtwebglplugin");
    widget.deselectComponent("qt.tools.qtcreator");
    widget.deselectComponent("qt.tools.qtcreatorcdbext");
    widget.deselectComponent("qt.tools.win32_mingw730");
    widget.deselectComponent("qt.tools.win64_mingw730");
    widget.deselectComponent("qt.tools.perl");
	widget.deselectComponent("qt.tools.vcredist_msvc2015_x86");
    gui.clickButton(buttons.NextButton);
}

Controller.prototype.LicenseAgreementPageCallback = function() {
    gui.currentPageWidget().AcceptLicenseRadioButton.setChecked(true);
    gui.clickButton(buttons.NextButton);
}

Controller.prototype.StartMenuDirectoryPageCallback = function() {
    gui.clickButton(buttons.NextButton);
}

Controller.prototype.ReadyForInstallationPageCallback = function()
{
    gui.clickButton(buttons.NextButton, 500);
}

Controller.prototype.FinishedPageCallback = function() {
    var checkBoxForm = gui.currentPageWidget().LaunchQtCreatorCheckBoxForm;
    if (checkBoxForm && checkBoxForm.launchQtCreatorCheckBox) {
        checkBoxForm.launchQtCreatorCheckBox.checked = false;
    }
    gui.clickButton(buttons.FinishButton);
}
